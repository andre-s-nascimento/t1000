package net.ddns.adambravo79.tmill;

import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Component;
import org.telegram.telegrambots.longpolling.interfaces.LongPollingUpdateConsumer;
import org.telegram.telegrambots.meta.api.methods.GetFile;
import org.telegram.telegrambots.meta.api.methods.send.SendMessage;
import org.telegram.telegrambots.meta.api.objects.Update;
import org.telegram.telegrambots.client.okhttp.OkHttpTelegramClient;
import org.telegram.telegrambots.meta.generics.TelegramClient;

import net.ddns.adambravo79.tmill.model.CastRecord;
import net.ddns.adambravo79.tmill.model.MovieResponse;
import net.ddns.adambravo79.tmill.service.AudioService;
import net.ddns.adambravo79.tmill.service.MovieService;

import java.util.List;
import java.util.stream.Collectors;
import java.nio.file.Files;
import java.nio.file.StandardCopyOption;
import java.io.File;

@Component
public class T1000BotService implements LongPollingUpdateConsumer {

    private final TelegramClient telegramClient;
    private final MovieService movieService;
    private final AudioService audioService;
    private final OpenAiService openAiService; // 1. ADICIONE ESTA LINHA

    @Value("${t1000.features.transcription-enabled}")
    private boolean transcriptionEnabled;

    // 2. ADICIONE O SERVIÇO NO CONSTRUTOR
    public T1000BotService(@Value("${bot.token}") String botToken,
            MovieService movieService,
            AudioService audioService,
            OpenAiService openAiService) {
        this.telegramClient = new OkHttpTelegramClient(botToken);
        this.movieService = movieService;
        this.audioService = audioService;
        this.openAiService = openAiService;
    }

    @Override
    public void consume(List<Update> updates) {
        System.out.println("DEBUG: Recebi " + updates.size() + " atualizações.");

        updates.forEach(update -> {
            if (update.hasMessage()) {
                var message = update.getMessage();
                long chatId = message.getChatId();

                // 1. Lógica para ÁUDIO / VOZ
                if (message.hasVoice() || message.hasAudio()) {
                    // ADICIONE ESTA VERIFICAÇÃO AQUI:
                    if (!transcriptionEnabled) {
                        System.out.println("DEBUG: Transcrição desativada via ENV. Ignorando áudio.");
                        enviarResposta(chatId, "🔇 A transcrição de áudio está temporariamente desativada.");
                        return;
                    }

                    System.out.println("DEBUG: Mensagem de voz recebida");
                    String fileId = message.hasVoice() ? message.getVoice().getFileId()
                            : message.getAudio().getFileId();
                    processarVoz(chatId, fileId);
                }

                // 2. Lógica de BUSCA (Texto)
                if (message.hasText()) {
                    System.out.println("DEBUG: Mensagem de texto recebida");
                    String text = message.getText(); // Correção: Pegar o texto da mensagem

                    if (text.toLowerCase().startsWith("t1000 buscar ")) {
                        String nomeFilme = text.substring(13).trim();
                        MovieResponse response = movieService.buscarFilme(nomeFilme);

                        if (response != null && !response.results().isEmpty()) {
                            var filmeBasico = response.results().get(0);

                            // Hidratação de Dados (Dia 2.5)
                            var filme = movieService.buscarDetalhesCompletos(filmeBasico.id());
                            var elencoRaw = movieService.buscarElenco(filme.id());
                            String ondeAssistir = movieService.buscarOndeAssistir(filme.id());

                            // Lógica da Bandeira
                            String bandeiras = "🌐";
                            if (filme.originCountry() != null && !filme.originCountry().isEmpty()) {
                                StringBuilder sb = new StringBuilder();
                                for (String code : filme.originCountry()) {
                                    String flag = getFlagEmoji(code);
                                    if (flag != null)
                                        sb.append(flag);
                                }
                                if (sb.length() > 0)
                                    bandeiras = sb.toString();
                            }

                            String elenco = elencoRaw.stream()
                                    .limit(5)
                                    .map(CastRecord::name)
                                    .collect(Collectors.joining(", "));

                            String ano = (filme.releaseDate() != null && filme.releaseDate().length() >= 4)
                                    ? filme.releaseDate().substring(0, 4)
                                    : "N/A";

                            String resposta = String.format(
                                    "🎬 *%s* (%s) %s\n" +
                                            "⭐ Nota: *%.1f*/10\n\n" +
                                            "📺 *Onde assistir:* %s\n\n" +
                                            "👥 *Elenco:* %s\n\n" +
                                            "%s\n\n" +
                                            "📖 *Sinopse:* %s",
                                    filme.title().toUpperCase(), ano, bandeiras,
                                    filme.voteAverage(), ondeAssistir, elenco,
                                    "https://image.tmdb.org/t/p/w500" + filme.posterPath(),
                                    filme.overview().length() > 1000 ? filme.overview().substring(0, 1000) + "..."
                                            : filme.overview());

                            enviarResposta(chatId, resposta);
                        } else {
                            enviarResposta(chatId, "❌ Não encontrei nenhum filme com esse nome.");
                        }
                    }
                }
            }
        });
    }

    private void processarVoz(long chatId, String fileId) {
        // Proteção redundante
        if (!transcriptionEnabled)
            return;

        try {
            // 1. Garante a pasta local
            java.io.File tempDir = new java.io.File("temp_audio");
            if (!tempDir.exists())
                tempDir.mkdir();

            // 2. Busca os metadados
            GetFile getFile = new GetFile(fileId);
            org.telegram.telegrambots.meta.api.objects.File telegramFile = telegramClient.execute(getFile);

            // 3. DOWNLOAD
            java.io.File fileTemporario = telegramClient.downloadFile(telegramFile);

            // 4. MOVE para pasta temp_audio
            String remotePath = telegramFile.getFilePath();
            String fileName = remotePath.substring(remotePath.lastIndexOf("/") + 1);
            java.io.File ogaFile = new java.io.File(tempDir, fileName);

            Files.move(fileTemporario.toPath(), ogaFile.toPath(), StandardCopyOption.REPLACE_EXISTING);

            // 5. CONVERSÃO E TRANSCRIÇÃO ASSÍNCRONA (Ajuste para não travar o Bot)
            audioService.converterParaWav(ogaFile)
                    .thenCompose(wavFile -> {
                        // Quando o WAV ficar pronto, chama a transcrição (também assíncrona/externa)
                        String textoBruto = openAiService.transcrever(wavFile);
                        enviarResposta(chatId, "🎙️ *Transcrição:* \n_" + textoBruto + "_");

                        // Passo 2: Refinamento (Llama 3 no Groq)
                        String textoRefinado = openAiService.refinarTexto(textoBruto);
                        enviarResposta(chatId, "✨ *Texto Refinado (Llama 3):* \n" + textoRefinado);

                        // Cleanup do WAV após a transcrição
                        wavFile.delete();
                        return java.util.concurrent.CompletableFuture.completedFuture(null);
                    })
                    .exceptionally(ex -> {
                        enviarResposta(chatId, "⚠️ Erro no processamento de áudio: " + ex.getMessage());
                        ex.printStackTrace();
                        return null;
                    })
                    .thenRun(ogaFile::delete); // Cleanup do OGA original ao final de tudo

        } catch (Exception e) {
            enviarResposta(chatId, "⚠️ Erro ao baixar arquivo: " + e.getMessage());
            e.printStackTrace();
        }
    }

    private String getFlagEmoji(String countryCode) {
        if (countryCode == null || countryCode.length() != 2)
            return null;
        try {
            int firstLetter = Character.codePointAt(countryCode.toUpperCase(), 0) - 0x41 + 0x1F1E6;
            int secondLetter = Character.codePointAt(countryCode.toUpperCase(), 1) - 0x41 + 0x1F1E6;
            return new String(Character.toChars(firstLetter)) + new String(Character.toChars(secondLetter));
        } catch (Exception e) {
            return null;
        }
    }

    private void enviarResposta(long chatId, String texto) {
        // Escapa caracteres que costumam quebrar o Markdown do Telegram em textos de IA
        String textoSeguro = texto
                .replace("_", "\\_")
                .replace("*", "\\*")
                .replace("[", "\\[")
                .replace("`", "\\`")
                .replace("(", "\\(")
                .replace(")", "\\)");

        SendMessage message = SendMessage.builder()
                .chatId(String.valueOf(chatId))
                .text(textoSeguro)
                .parseMode("Markdown")
                .build();
        try {
            telegramClient.execute(message);
        } catch (Exception e) {
            System.err.println("Erro ao enviar: " + e.getMessage());
            // Se ainda assim falhar, envia como texto puro para não perder a informação
            try {
                telegramClient.execute(new SendMessage(String.valueOf(chatId), texto));
            } catch (Exception e2) {
                e2.printStackTrace();
            }
        }
    }
}
