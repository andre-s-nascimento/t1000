package net.ddns.adambravo79.tmill;

import org.springframework.http.MediaType;
import org.springframework.http.client.MultipartBodyBuilder;
import org.springframework.http.client.SimpleClientHttpRequestFactory;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.core.io.FileSystemResource;
import org.springframework.stereotype.Service;
import org.springframework.web.client.RestClient;

import net.ddns.adambravo79.tmill.model.TranscriptionResponse;

import java.io.File;
import java.util.List;
import java.util.Map;

@Service
public class OpenAiService {

    private final RestClient restClient;

    // AQUI: Declaramos a variável e injetamos o valor do application.properties ou
    // ENV
    public OpenAiService(@Value("${groq.api.key:none}") String groqKey) {
        var factory = new SimpleClientHttpRequestFactory();
        factory.setConnectTimeout(5000);
        factory.setReadTimeout(30000);

        this.restClient = RestClient.builder()
                .requestFactory(factory)
                // Remova o "/openai/v1" da baseUrl se ele estiver causando conflito
                .baseUrl("https://api.groq.com")
                .defaultHeader("Authorization", "Bearer " + groqKey)
                .build();
    }

    public String refinarTexto(String textoBruto) {

        if (textoBruto.length() > 5000) {
            return textoBruto + "\n\n⚠️ (Texto muito longo para refinamento automático)";
        }

        try {
            // Usamos o Llama 3 do Groq, que é excelente para processamento de linguagem
            // natural
            var payload = java.util.Map.of(
                    "model", "llama-3.1-8b-instant",
                    "messages", java.util.List.of(
                            java.util.Map.of("role", "system", "content",
                                    "Você é um assistente de revisão. Sua tarefa é transformar uma transcrição bruta em um texto fluido e legível. "
                                            +
                                            "Instruções: 1. Corrija pontuação e gramática. 2. Remova vícios de fala (pô, né, ah, tipo). "
                                            +
                                            "3. Mantenha o vocabulário original e o tom informal. 4. Retorne apenas o texto corrigido, sem introduções."),
                            java.util.Map.of("role", "user", "content", textoBruto)),
                    "temperature", 0.5 // Menor temperatura para manter o texto fiel ao original
            );

            var response = restClient.post()
                    .uri("/openai/v1/chat/completions") // O Groq usa este path para compatibilidade
                    .contentType(org.springframework.http.MediaType.APPLICATION_JSON)
                    .body(payload)
                    .retrieve()
                    .body(java.util.Map.class);

            // Extração segura do conteúdo da resposta
            var choices = (java.util.List<java.util.Map<String, Object>>) response.get("choices");
            var message = (java.util.Map<String, Object>) choices.get(0).get("message");

            return (String) message.get("content");

        } catch (Exception e) {
            return "⚠️ Erro ao refinar com Llama-3: " + e.getMessage();
        }
    }

    public String transcrever(File wavFile) {
        try {
            MultipartBodyBuilder builder = new MultipartBodyBuilder();
            builder.part("file", new org.springframework.core.io.FileSystemResource(wavFile));
            builder.part("model", "whisper-large-v3");

            // AQUI: O .body(TranscriptionResponse.class) mapeia o JSON para o seu record
            TranscriptionResponse response = restClient.post()
                    .uri("/openai/v1/audio/transcriptions")
                    .contentType(org.springframework.http.MediaType.MULTIPART_FORM_DATA)
                    .body(builder.build())
                    .retrieve()
                    .body(TranscriptionResponse.class);

            return (response != null && response.text() != null)
                    ? response.text()
                    : "Silêncio absoluto... nada detectado.";

        } catch (Exception e) {
            return "⚠️ Falha na unidade de voz: " + e.getMessage();
        }
    }
}
